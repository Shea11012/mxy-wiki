## 选择排序

````ad-example
title: 选择排序

~~~go
// Selection 选择排序
func Selection(array []int) {
	for i := 0; i < len(array)-1; i++ {
		minIndex := i
		for j := i + 1; j < len(array); j++ {
			if array[j] < array[minIndex] {
				minIndex = j
			}
		}

		if i != minIndex {
			swap(array, i, minIndex)
		}
	}
}
````
