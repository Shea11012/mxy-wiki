## 希尔排序

```go
func Shell(array []int) {
	// 希尔排序增量
	d := len(array)
	for d > 1 {
		// 希尔排序增量，每次折半
		d = d / 2
		for i := 0; i < d; i++ {
			for j := i + d; j < len(array); j = j + d {
				temp := array[j]
				var k int
				for k = j - d; k >= 0 && array[k] > temp; k = k - d {
					array[k+d] = array[k]
				}
				array[k+d] = temp
			}
		}
	}
}
```
