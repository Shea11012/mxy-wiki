func binarySearch(arr []int,target int) int {
	start := 0
	end := len(arr) - 1
	var mid int

	for start <= end {
		// (start + end) / 2 在 start 和 end 是当前类型的最大值时会溢出
		mid = start + ((end - start) / 2)
		if arr[mid] == target {
			return mid
		}

		if arr[mid] > target {
			end = mid - 1
		} else {
			start = mid + 1
		}
	}

	return -1
}
